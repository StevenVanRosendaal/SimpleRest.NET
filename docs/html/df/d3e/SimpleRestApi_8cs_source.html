154 async Task RunMiddleWare(SimpleRestRequest request, SimpleRestResponse response)
155 {
156 Dictionary<UriTemplateMatch, SimpleRestMap> matches = m_Middleware
157 .Where(
158 m => m.Pattern.Match(new System.Uri(request.Endpoint, UriKind.Relative)) != null

161 m => m.Pattern.Match(new System.Uri(request.Endpoint, UriKind.Relative)), m => m

163 matches.Values.ToList().Dump();
164 OnHandleRequestStack?.Invoke(this, request, response, matches);
165
166
167 foreach (KeyValuePair<UriTemplateMatch, SimpleRestMap> match in matches)
168 {
169 if (response.HasCompleted)
170 return;

172 UriTemplateMatch uriTemplateMatch = match.Key;
173
174 if (map.Method == SimpleRestMethod.ANY || map.Method == request.Method)
175 {
176 OnRequestMatch?.Invoke(this, request, response, uriTemplateMatch, map);
177 ApplyUriParams(uriTemplateMatch, request);
178 OnApplyUriParams?.Invoke(this, request, response, uriTemplateMatch, map);
179 OnBeforeRunMiddleware?.Invoke(this, request, response, uriTemplateMatch, map);
180 await map.Middleware.Invoke(request, response);
181 OnRunMiddleware?.Invoke(this, request, response, uriTemplateMatch, map);

191 void ApplyUriParams(UriTemplateMatch match, SimpleRestRequest request)
192 {
193 match.Dump();
194 Dictionary<string, object?> paramsToAdd = [];
195 foreach (string key in match.Bindings.Keys)
196 {
197 paramsToAdd.Add(key, match.Bindings[key].Value);
198 object? converted = null;
199 try
200 {
201 converted = JsonConvert.DeserializeObject(match.Bindings[key].Value.ToString() ?? "null");
202 }
203 catch (JsonException je)
204 {
205 string stringifiedObject = $"\"{match.Bindings[key].Value.ToString()}\"";
206 converted = JsonConvert.DeserializeObject(stringifiedObject ?? "null");
207
208 }
209 converted = ApplyIntType(converted);
210 paramsToAdd[key] = converted;
211 }
212 request.Params.NonDistructiveUnion(paramsToAdd);
213 request.Params.Dump();
214
