131 public async Task Start(Action<int, string>? OnStartup = null)
132 {
133 try
134 {
135
136 m_Listener.Start();
137 MapHandlers();
138
139 OnServerStart?.Invoke(this);
140 OnStartup?.Invoke(m_Port, m_Listener.Prefixes.First().Replace("*", "localhost"));
141 while (true)
142 {
143 try
144 {
145 HttpListenerContext context = await m_Listener.GetContextAsync();
146 OnBeforeRequestCreate?.Invoke(this);
147 SimpleRestRequest request = SimpleRestRequest.FromHttpListenerContext(context, m_EndpointFormatter);
148 OnRequestCreate?.Invoke(this, request);
149 OnBeforeResponseCreate?.Invoke(this, request);
150 SimpleRestResponse response = new SimpleRestResponse(context.Response, m_ResponseTypeParser);
151 OnResponseCreate?.Invoke(this, request, response);
152
153 m_Logger.Log(request);
154 OnLog?.Invoke(this, request);
155 await RunMiddleWare(request, response);
156 OnBeforeRequestEnd?.Invoke(this, request, response);
157 context.Response.Close();
158 OnRequestEnd?.Invoke(this, request, response);

160 }
161 catch (Exception e)

163 Console.WriteLine("Something went wrong while getting request: " + e.Message + " " + e.StackTrace);

174 async Task RunMiddleWare(SimpleRestRequest request, SimpleRestResponse response)
175 {
176 Dictionary<UriTemplateMatch, SimpleRestMap> matches = m_Middleware
177 .Where(
178 m => m.Pattern.Match(new System.Uri(request.Endpoint, UriKind.Relative)) != null
179 )
180 .ToDictionary(
181 m => m.Pattern.Match(new System.Uri(request.Endpoint, UriKind.Relative)), m => m
182 );
183 matches.Values.ToList().Dump();
184 OnHandleRequestStack?.Invoke(this, request, response, matches);
185
186
187 foreach (KeyValuePair<UriTemplateMatch, SimpleRestMap> match in matches)
188 {
189 if (response.HasCompleted)
190 return;
191 SimpleRestMap map = match.Value;
192 UriTemplateMatch uriTemplateMatch = match.Key;
193
194 if (map.Method == SimpleRestMethod.ANY || map.Method == request.Method)
195 {
196 OnRequestMatch?.Invoke(this, request, response, uriTemplateMatch, map);
197 ApplyUriParams(uriTemplateMatch, request);
198 OnApplyUriParams?.Invoke(this, request, response, uriTemplateMatch, map);
199 OnBeforeRunMiddleware?.Invoke(this, request, response, uriTemplateMatch, map);
200 await map.Middleware.Invoke(request, response);
201 OnRunMiddleware?.Invoke(this, request, response, uriTemplateMatch, map);
202

211 void ApplyUriParams(UriTemplateMatch match, SimpleRestRequest request)
212 {
213 match.Dump();
214 Dictionary<string, object?> paramsToAdd = [];
215 foreach (string key in match.Bindings.Keys)
216 {
217 paramsToAdd.Add(key, match.Bindings[key].Value);
218 object? converted = null;
219 try
220 {
221 converted = JsonConvert.DeserializeObject(match.Bindings[key].Value.ToString() ?? "null");
222 }
223 catch (JsonException je)
224 {
225 string stringifiedObject = $"\"{match.Bindings[key].Value.ToString()}\"";
226 converted = JsonConvert.DeserializeObject(stringifiedObject ?? "null");
227
228 }
229 converted = ApplyIntType(converted);
230 paramsToAdd[key] = converted;
231 }
232 request.Params.NonDistructiveUnion(paramsToAdd);
233 request.Params.Dump();
234
235 }
