167 public async Task Start(Action<int, string>? OnStartup = null)
168 {
169 try
170 {
171 m_Listener.Start();
172 MapHandlers();
173
174 OnServerStart?.Invoke(this);
175 OnStartup?.Invoke(m_Port, m_Listener.Prefixes.First().Replace("*", "localhost"));
176 HasStarted = true;
177 Disposed = false;
178 while (!Disposed)
179 {
180 try
181 {
182 HttpListenerContext context = await m_Listener.GetContextAsync();
183 OnBeforeRequestCreate?.Invoke(this);
184
185 SimpleRestRequest request = SimpleRestRequest.FromHttpListenerContext(
186 context,
187 m_EndpointFormatter
188 );
189 OnRequestCreate?.Invoke(this, request);
190 OnBeforeResponseCreate?.Invoke(this, request);
191 SimpleRestResponse response = new SimpleRestResponse(
192 context.Response,
193 m_ResponseTypeParser
194 );
195 OnResponseCreate?.Invoke(this, request, response);
196
197 m_Logger.Log(request);
198 OnLog?.Invoke(this, request);
199 if (request.Method == SimpleRestMethod.OPTIONS)
200 {
201 //TODO handle options request
202 }
203 else
204 {
205 await RunMiddleWare(request, response);
206
207 OnBeforeRequestEnd?.Invoke(this, request, response);
208 context.Response.Close();
209 OnRequestEnd?.Invoke(this, request, response);
210 }
211 }
212 catch (ObjectDisposedException ode)
213 {
214 Console.WriteLine("Stopping server due to dead listener");
215 Stop();
216 break;
217 }
218 catch (Exception e)
219 {
220 Console.WriteLine(
221 "Something went wrong while getting request: "
222 + e.Message
223 + " "
224 + e.StackTrace
225 );
226 }
227 }
228 }
229 catch (Exception e)
230 {
231 Console.WriteLine("Failure to open server: " + e.Message);
232 }

235 async Task RunMiddleWare(SimpleRestRequest request, SimpleRestResponse response)
236 {
237 Dictionary<UriTemplateMatch, SimpleRestMap> matches = m_Middleware
238 .Where(m => m.Pattern.Match(new System.Uri(request.Endpoint, UriKind.Relative)) != null)
239 .ToDictionary(
240 m => m.Pattern.Match(new System.Uri(request.Endpoint, UriKind.Relative)),
241 m => m
242 );
243 matches.Values.ToList().Dump();
244 OnHandleRequestStack?.Invoke(this, request, response, matches);
245
246 foreach (KeyValuePair<UriTemplateMatch, SimpleRestMap> match in matches)
247 {
248 if (response.HasCompleted)
249 return;
250 SimpleRestMap map = match.Value;
251 UriTemplateMatch uriTemplateMatch = match.Key;
252
253 if (map.Method == SimpleRestMethod.ANY || map.Method == request.Method)
254 {
255 OnRequestMatch?.Invoke(this, request, response, uriTemplateMatch, map);
256 ApplyUriParams(uriTemplateMatch, request);
257 OnApplyUriParams?.Invoke(this, request, response, uriTemplateMatch, map);
258 OnBeforeRunMiddleware?.Invoke(this, request, response, uriTemplateMatch, map);
259 await map.Middleware.Invoke(request, response);
260 OnRunMiddleware?.Invoke(this, request, response, uriTemplateMatch, map);
261 }
262 }

265 void ApplyUriParams(UriTemplateMatch match, SimpleRestRequest request)
266 {
267 match.Dump();
268 Dictionary<string, object?> paramsToAdd = new Dictionary<string, object?>();
269 foreach (string key in match.Bindings.Keys)
270 {
271 paramsToAdd.Add(key, match.Bindings[key].Value);
272 object? converted = null;
273 try
274 {
275 converted = JsonConvert.DeserializeObject(
276 match.Bindings[key].Value.ToString() ?? "null"
277 );
278 }
279 catch (JsonException je)
280 {
281 string stringifiedObject = $"\"{match.Bindings[key].Value.ToString()}\"";
282 converted = JsonConvert.DeserializeObject(stringifiedObject ?? "null");
283 }
284 converted = ApplyIntType(converted);
285 paramsToAdd[key] = converted;
286 }
287 request.Params.NonDistructiveUnion(paramsToAdd);
288 request.Params.Dump();
289 }
