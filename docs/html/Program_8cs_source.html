190 public static SimpleRestRequest FromHttpListenerContext(HttpListenerContext listener)
191 {
192 HttpListenerRequest contextRequest = listener.Request;
193
194 SimpleRestRequest request = new SimpleRestRequest();
195 request.Query = contextRequest.QueryString.AllKeys.ToDictionary(k => k, k => contextRequest.QueryString[k].SafeDeserialize(contextRequest.QueryString[k]));
196 request.Body = new SimpleRestRequestBody(contextRequest);
197 request.Headers = contextRequest.Headers?.AllKeys.ToDictionary(k => k, k => contextRequest.Headers[k]);
198 request.ContentType = contextRequest.ContentType;
199 request.ContentLength = contextRequest.ContentLength64;
200 request.Endpoint = contextRequest.Url?.AbsolutePath;
201 request.Method = Enum.Parse<SimpleRestMethod>(contextRequest.HttpMethod);
202 // Get the raw URL and extract the path
203 string rawUrl = contextRequest.RawUrl;
204 string path = rawUrl.Split('?')[0]; // Exclude any query string
205 string[] pathSegments = path.Split(new[] { '/' }, StringSplitOptions.RemoveEmptyEntries); // Split by slash
206 request.Path = pathSegments;
207
208 return request;
209 }
